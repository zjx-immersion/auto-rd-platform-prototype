**问题**: PI周期不固定，里程碑经常错位

**解决方案**:
```typescript
// 固定12周节奏
function calculatePIDates(previousPI) {
  const start = dayjs(previousPI.endDate).add(1, 'day') // 无间隙
  const end = start.add(12, 'week').subtract(1, 'day')  // 固定12周
  return { start, end }
}

// 里程碑对齐算法（±7天）
function checkMilestoneAlignment(piEndDate, milestoneDate) {
  const absDiff = Math.abs(dayjs(milestoneDate).diff(piEndDate, 'day'))
  
  if (absDiff <= 7) return 'PERFECT'      // 完美对齐
  if (absDiff <= 14) return 'ACCEPTABLE'  // 可接受对齐
  return 'MISALIGNED'                     // 不对齐
}

// 3种调整方案建议
function generateSuggestions(daysDiff) {
  return [
    { type: 'ADJUST_PI', risk: 'MEDIUM' },
    { type: 'ADJUST_MILESTONE', risk: 'HIGH' },
    { type: 'NO_LINK', risk: 'LOW', recommended: true } ⭐
  ]
}
```
