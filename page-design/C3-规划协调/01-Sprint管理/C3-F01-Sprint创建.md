```sql
CREATE TABLE sprints (
  sprint_id UUID PRIMARY KEY,
  pi_id UUID NOT NULL REFERENCES pis(pi_id),
  project_id UUID NOT NULL REFERENCES projects(project_id),
  sprint_number VARCHAR(50) NOT NULL,
  sprint_name VARCHAR(200) NOT NULL,
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  duration_weeks INTEGER NOT NULL DEFAULT 2,
  sprint_type VARCHAR(20) NOT NULL, -- development, ip
  goal_description TEXT,
  success_criteria TEXT,
  team_id UUID NOT NULL REFERENCES teams(team_id),
  scrum_master_id UUID NOT NULL REFERENCES users(user_id),
  product_owner_id UUID REFERENCES users(user_id),
  available_person_days INTEGER,
  holiday_days INTEGER DEFAULT 0,
  leave_days DECIMAL(5,2) DEFAULT 0,
  training_days DECIMAL(5,2) DEFAULT 0,
  other_days DECIMAL(5,2) DEFAULT 0,
  actual_capacity INTEGER,
  status VARCHAR(20) NOT NULL, -- draft, created, in_progress, completed, closed, cancelled
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  created_by UUID REFERENCES users(user_id),
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_by UUID REFERENCES users(user_id),
  UNIQUE(pi_id, sprint_number)
);

CREATE INDEX idx_sprints_pi ON sprints(pi_id);
CREATE INDEX idx_sprints_project ON sprints(project_id);
CREATE INDEX idx_sprints_team ON sprints(team_id);
CREATE INDEX idx_sprints_status ON sprints(status);
```
