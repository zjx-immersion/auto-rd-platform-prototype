### 7.2 Sprintå®¹é‡ç»Ÿè®¡è§†å›¾

```sql
CREATE VIEW sprint_capacity_stats AS
SELECT
  s.sprint_id,
  s.sprint_number,
  s.actual_capacity AS available_capacity,
  COALESCE(SUM(sb.planned_hours), 0) AS planned_hours,
  s.actual_capacity - COALESCE(SUM(sb.planned_hours), 0) AS remaining_capacity,
  CASE
    WHEN s.actual_capacity > 0
    THEN COALESCE(SUM(sb.planned_hours), 0) / s.actual_capacity
    ELSE 0
  END AS utilization_rate,
  COALESCE(SUM(sb.planned_story_points), 0) AS planned_story_points,
  COUNT(sb.id) AS total_mrs
FROM sprints s
LEFT JOIN sprint_backlog sb ON s.sprint_id = sb.sprint_id
GROUP BY s.sprint_id, s.sprint_number, s.actual_capacity;
```
