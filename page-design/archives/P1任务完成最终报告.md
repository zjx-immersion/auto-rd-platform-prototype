```typescript
function predictCompletion(
  remainingSP: number,
  historicalVelocities: number[]
): Forecast {
  const avgVelocity = average(historicalVelocities);
  const stdDev = standardDeviation(historicalVelocities);
  
  return {
    realistic: remainingSP / avgVelocity,
    optimistic: remainingSP / (avgVelocity + stdDev),
    pessimistic: remainingSP / (avgVelocity - stdDev),
    confidence: 1 - (stdDev / avgVelocity)
  };
}
```
